def search_knowledge(query: str) -> str:
    """Поиск по базе знаний."""
    query_lower = query.lower()
    results = []

    def search_dict(d, path=""):
        if isinstance(d, dict):
            for key, value in d.items():
                key_str = str(key).lower()
                if any(q in key_str for q in query_lower.split()):
                    results.append(f"[{path}/{key}]")
                search_dict(value, f"{path}/{key}")
        elif isinstance(d, (str, list)):
            content = str(d).lower()
            if any(q in content for q in query_lower.split()):
                results.append(path)

    search_dict(KNOWLEDGE_BASE)
    return results


def get_author_info(name: str) -> dict | None:
    """Получить информацию об авторе."""
    for section in KNOWLEDGE_BASE.values():
        if isinstance(section, dict):
            for key, value in section.items():
                if name.lower() in key.lower() or key.lower() in name.lower():
                    return {"name": key, "info": value}
    return None


def get_nbp_history() -> str:
    """Получить историю НБП (признана экстремистской в РФ)."""
    hist = KNOWLEDGE_BASE["история_нбп"]
    text = f"ИСТОРИЯ НБП (признана экстремистской в РФ)\n\n"
    text += f"ОСНОВАНИЕ:\n{hist['основание']}\n\n"
    text += f"ЗАПРЕТ:\n{hist['запрет']}\n\n"
    text += "АКЦИИ НБП (признана экстремистской в РФ):\n"
    for period, actions in hist["акции"].items():
        text += f"\n{period.upper()}:\n"
        for action in actions:
            text += f"• {action}\n"
    return text
